public class  three { // three라는 클래스 생성

	public static void main(String[] args) { // 메인메소드 실행
	
		for(int x=0;x<=10;x++) { // x는 0에서 부터 10보다 작거나 같을 때 까지 1씩 증가 하다 조건에 맞으면 반복문 탈출 그렇지 않으면 다시 반복
			
			for(int y=0;y<=10;y++) { //y는 0에서부터 10보다 작거나 같을 때 까지 1씩 증가 하다 조건에 맞으면 x반복문으로 탈출 그렇지 않으면 다시 반복
				
				if((3*x +10*y==100)) { // 만약 3*x +10*y의 식이 100과 같다면
					
					System.out.println("("+" "+x+","+" "+y+")"); // 모니터에 s, y값 출력. 그렇지 않으면 다시 반복
				}
			}
		}
	}

}
